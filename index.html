Integrate in Minutes

payment_flow.ts 
import { PulseClient } from '@pulsepay/sdk';

// Initialize Client
const client = new PulseClient({
  network: 'mainnet-beta',
  apiKey: process.env.PULSE_KEY
});

// Execute Instant Payment
const tx = await client.transfer({
  to: 'user_wallet_address',
  amount: 50.00,
  token: 'USDC',
  memo: 'Service Payment'
});

console.log('Settled:', tx.signature);
